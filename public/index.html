Draw Pixel 

// Minimal example of sending a "draw" transaction
await near.sendTx({
  receiverId: "berryclub.ek.near",
  actions: [
    near.actions.functionCall({
      methodName: "draw",
      args: {
        pixels: [{
          x: 10,  // example x
          y: 20,  // example y
          color: 65280, // green in decimal
        }],
      },
      gas: $$`100 Tgas`,
      deposit: "0",
    }),
  ],
});
